## Example usage {#exampleUsage}

<Tabs>
    <TabItem value="usage" label="Using plugins" default>
        ```js
        const { Manager } = require("magmastream");
        const MyPlugin = require("my-magmastream-plugin");

        const manager = new Manager({
            ...ManagerOptions,
            plugins: [new MyPlugin({ foo: "bar" })],
        });
        ```
    </TabItem>
    <TabItem value="writing" label="Writing plugins" default>
        ```js
        const { Plugin } = require("magmastream");

        module.exports = class MyPlugin extends Plugin {
            constructor(options) {
                super();
                this.options = options; // will be { foo: 'bar' }
            }

            load(manager) {}
        };
        ```
    </TabItem>
